# 1、解压并修改名称

解压 Spark 到 /opt/module 并修改名称

[laowang@hadoop102 software]$ tar -zxvf spark-3.0.0-bin-hadoop3.2.tgz -C /opt/module/
[laowang@hadoop102 software]$ cd /opt/module
[laowang@hadoop102 module]$ mv spark-3.0.0-bin-hadoop3.2 spark-on-hive

# 2、创建配置文件

在 /opt/module/spark-on-hive/conf 创建 hive-site.xml

[laowang@hadoop102 module]$ cd /opt/module/spark-on-hive
[laowang@hadoop102 spark-on-hive]$ vim conf/hive-site.xml

<?xml version="1.0"?>
<?xml-stylesheet type="text/xsl" href="configuration.xsl"?>
<configuration>
    <property>
        <name>hive.metastore.warehouse.dir</name>
        <value>/user/hive/warehouse</value>
    </property>
    <property>
        <name>hive.metastore.uris</name>
        <value>thrift://hadoop102:9083</value>
    </property>
</configuration>

# 3、启动 Hive 的 MetaStore 服务

[laowang@hadoop102 spark-on-hive]$ cd /opt/module/hive-3.1.2/
[laowang@hadoop102 hive-3.1.2]$ bin/hive --service metastore
# 后台启动
[laowang@hadoop102 hive-3.1.2]$ bin/hive --service metastore 2>&1 >> /var/log.log &

# 4、启动 spark-sql 命令

[laowang@hadoop102 hive-3.1.2]$ cd /opt/module/spark-on-hive
[laowang@hadoop102 spark-on-hive]$ bin/spark-sql
# yarn 模式
[laowang@hadoop102 spark-on-hive]$ bin/spark-sql --master yarn

# 5、Spark 的 ThriftServer 服务

使用 Spark 的 ThriftServer 服务，类似于 Hive 的 HiveService2，启动后可以使用 DBeaver等工具连接，使用 JDBC 操作数据。

在 hive-site.xml 添加配置

[laowang@hadoop102 spark-on-hive]$ vim conf/hive-site.xml

    <property>
        <name>hive.server2.thrift.port</name>
        <value>10000</value>
    </property>
    <property>
        <name>hive.server2.thrift.bind.host</name>
        <value>hadoop102</value>
    </property>

启动 ThriftServer 服务

[laowang@hadoop102 spark-on-hive]$ sbin/start-thriftserver.sh
# yarn 模式
[laowang@hadoop102 spark-on-hive]$ sbin/start-thriftserver.sh --master yarn
